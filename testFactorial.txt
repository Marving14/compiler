program factorial;

Int module fact(num):
    %%write(num)
    var Int acum=1; 
    
    for i=1 to num+1 do:
        acum = acum*i
        %%write(acum)
    endFor;

    %%write(acum)
    return acum;
end

Int module recursivo(num):
    if num >= 1 then:
        return num * recursivo(num-1)
    else:
        return 1
    end
end
