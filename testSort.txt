module sort(list):
    %%  TEST SORT

    var sorted = list; 
    var n = len(sorted);

    for a = 0 to n step 1 do:
        for b = 0 to n-1 step 1 do:
            var cont = b+1;

            if sorted/b > sorted/cont then:
                var temp = sorted/b;
                set(sorted, b, sorted/cont);
                set(sorted, cont, temp);
            end
        endFor
    endFor
    return sorted;
end

    var arr = [6,7,3,51,9,1];

    write(sort(arr))
